#include<iostream>
using namespace std;
int const col=16;
int const row =16;
void printarray(int a[row][col],int row,int col)
{
	for(int i=0;i<row;i++)
	{
	   for(int j=0;j<col;j++)
	   {
		   cout<<a[i][j]<<"\t";
	   }
	   cout<<endl;
	}

}
void multiplicationtarr(int x[row][col],int y[row][col],int z[row][col],int row,int col)
{
	for(int i=0;i<row;i++)
	{
	   for(int j=0;j<col;j++)
	   {
		   z[i][j]=x[i][j]*y[i][j];
		   cout<<z[i][j]<<"\t";
	   }
	   cout<<endl;
	}
}
